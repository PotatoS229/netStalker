fn calculate_max_attempts(min: u32, max: u32) -> u32 {

fn game_loop(secret_number: u32, max_attempts: u32, min: u32, max: u32) {
    let mut attempts = 0;
    let mut previous_guesses = Vec::new();
    
    loop {
        attempts += 1;
        
        // –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø—Ä–µ–≤—ã—à–µ–Ω–∏–µ –ª–∏–º–∏—Ç–∞ –ø–æ–ø—ã—Ç–æ–∫

            println!("üíÄ –ò–ì–†–ê –û–ö–û–ù–ß–ï–ù–ê! –í—ã –∏—Å—á–µ—Ä–ø–∞–ª–∏ –≤—Å–µ –ø–æ–ø—ã—Ç–∫–∏.");
            println!("üíÄ –ó–∞–≥–∞–¥–∞–Ω–Ω–æ–µ —á–∏—Å–ª–æ –±—ã–ª–æ: {}", secret_number);
            println!("{}", "üíÄ".repeat(30));
            break;
        }
        
        println!("\n{}", "‚îÄ".repeat(30));
        println!("üìä –ü–æ–ø—ã—Ç–∫–∞ {}/{}", attempts, max_attempts);
        
        if !previous_guesses.is_empty() {
            println!("üìù –ü—Ä–µ–¥—ã–¥—É—â–∏–µ –ø–æ–ø—ã—Ç–∫–∏: {:?}", previous_guesses);
        }
        
        // –ü–æ–ª—É—á–µ–Ω–∏–µ –≤–≤–æ–¥–∞ –æ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
        let guess = match get_user_input(min, max) {
            Some(num) => num,
            None => {
                attempts -= 1; // –ù–µ –∑–∞—Å—á–∏—Ç—ã–≤–∞–µ–º –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—É—é –ø–æ–ø—ã—Ç–∫—É
                continue;
            }
        };
        
        previous_guesses.push(guess);
        
        // –°—Ä–∞–≤–Ω–µ–Ω–∏–µ –∏ –≤—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
        match guess.cmp(&secret_number) {
            Ordering::Less => {
                println!("üìâ –°–ª–∏—à–∫–æ–º –º–∞–ª–µ–Ω—å–∫–æ–µ!");

            Ordering::Equal => {

    }

fn get_user_input(min: u32, max: u32) -> Option<u32> {

            None

    }
}

fn give_hint(secret: u32, guess: u32, attempts: u32, max_attempts: u32) {
    if attempts >= max_attempts / 2 {
        let difference = if secret > guess {
            secret - guess
        } else {
            guess - secret
        };
        
        if difference <= 10 {
            println!("üí° –ì–æ—Ä—è—á–æ! –û—á–µ–Ω—å –±–ª–∏–∑–∫–æ!");

            println!("üí° –¢–µ–ø–ª–æ!");
        } else if difference <= 50 {
            println!("üí° –ü—Ä–æ—Ö–ª–∞–¥–Ω–æ...");
        } else {
            println!("üí° –•–æ–ª–æ–¥–Ω–æ!");
        }
        
        // –ü–æ–¥—Å–∫–∞–∑–∫–∞ –æ —á–µ—Ç–Ω–æ—Å—Ç–∏ –ø–æ—Å–ª–µ 1/3 –ø–æ–ø—ã—Ç–æ–∫
        if attempts == max_attempts / 3 {

            }
        }
    }
}

fn calculate_score(attempts: u32, max_attempts: u32) -> u32 {
    let efficiency = (max_attempts as f64 - attempts as f64) / max_attempts as f64;
